► SYSTEM REQUIREMENTS ────────────────────── [✓] Go 1.21+ [✓] tmux 3.0+ [✓] Terminal with 256-color support [ ] Claude API key (for AI features) [ ] Ollama (optional, for local models) ► INSTALLATION ────────────── # Clone the repository 
git clone https://github.com/orcai/orcai
cd orcai
make build
./bin/orcai
# Or install directly with go install 
go install github.com/orcai/orcai@latest
► FIRST LAUNCH ───────────── # orcai boots into a BubbleTea TUI inside a tmux session orcai
orcai creates a tmux session named "orcai" and attaches to it. The sidebar on the left shows active sessions and plugin nodes. The prompt builder panel lives on the right. ► KEYBINDINGS ──────────── 
Ctrl+;        Open chord menu
Ctrl+; s      New tmux session
Ctrl+; p      Open prompt builder
Ctrl+; g      Plugin gallery
Ctrl+; q      Quit orcai
?             Show help overlay
Tab           Cycle panels
► CONFIGURATION ────────────── # Config lives at ~/.config/glitch/config.yaml 
providers:
  claude:
    api_key: "${ANTHROPIC_API_KEY}"
    model: claude-opus-4-5
  ollama:
    base_url: "http://localhost:11434"
    model: llama3.2

tmux:
  session_prefix: "orcai-"
  default_layout: tiled

plugins:
  dir: "~/.config/glitch/plugins"
  autoload: true
► ENVIRONMENT VARIABLES ─────────────────────── 
export ANTHROPIC_API_KEY="sk-ant-..."    # Claude API
export OLLAMA_HOST="http://localhost:11434"  # Ollama base
export GLITCH_CONFIG="~/.config/orcai"   # Config dir override
export GLITCH_LOG_LEVEL="debug"          # Logging verbosity
► YOUR FIRST PLUGIN ─────────────────── # Wrap any CLI tool with a sidecar YAML — no code required 
# ~/.config/glitch/plugins/jq.yaml
name: jq
description: JSON processor
command: jq
args: ["-r", "."]
input_schema: '{"type":"string","description":"JSON input"}'
output_schema: '{"type":"string","description":"Processed output"}'
# Drop the YAML in the plugins dir and reload 
mkdir -p ~/.config/glitch/plugins
cp jq.yaml ~/.config/glitch/plugins/
# In orcai: Ctrl+; g  →  Reload Plugins
